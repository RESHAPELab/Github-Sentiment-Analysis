      hasNextPage

    repositoryCount
    nodes {
      ... on Repository {
        owner {
          login
          __typename
        }
        name
        createdAt
        pushedAt
        isMirror
        stargazers {
          totalCount
        }
        issues {
          totalCount
        }
        pullRequests(first: 10) {
          totalCount
          nodes {
            title
            createdAt
            number
            closed
            author {
              login
              __typename
            }
            comments(first: 10) {
              edges {
                node {
                  author{
                    login
                    __typename
                  }
                  bodyText
                }
              }
            }
            reviewThreads(first: 10) {
              edges {
                node {
                  comments(first: 10) {
                    nodes {

                        login
                        __typename
